*** Requires
- git
- curl
- wget
*** Install
#+BEGIN_SRC 
curl -L https://raw.githubusercontent.com/damon-kwok/oh-my-env/master/install.sh | bash
#+END_SRC

*** Uninstall
#+BEGIN_SRC shell
ome uninstall self
#+END_SRC
or
#+BEGIN_SRC shell
curl -L https://raw.githubusercontent.com/damon-kwok/oh-my-env/master/uninstall.sh | bash
#+END_SRC
or
#+BEGIN_SRC shell
rm -rf ~/.oh-my-env/
rm -rf ~/.ome_local/
#+END_SRC
*** Usage
- init
#+BEGIN_SRC shell
ome init
#+END_SRC
- install
#+BEGIN_SRC shell

ome install jvm

ome install rust
ome install python
ome install ruby
ome install perl
ome install erlang
ome install nim
#+END_SRC
- uninstall
#+BEGIN_SRC shell
ome uninstall cc
#+END_SRC
- update
#+BEGIN_SRC shell

#+END_SRC
